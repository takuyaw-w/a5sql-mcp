async function assertPublishedPackageClientMatrix(client, options) {
  const { adversarialRoot, hostileSqlPath } = options;
  const rawSecrets = [
    { label: "sql password literal", value: "published-fixture-password" },
    { label: "bearer token", value: "published-fixture-bearer" },

async function callToolStructured(client, name, args) {
  const result = await client.callTool({ name, arguments: args });
  const structuredContent = result.structuredContent;
  if (
    !structuredContent ||
    typeof structuredContent !== "object" ||
    Array.isArray(structuredContent)
  ) {
    throw new Error(`Published-style matrix ${name} did not return object structuredContent.`);
  }
  return structuredContent;
}

function assertObjectIncludes(label, actual, expected) {
  for (const [key, expectedValue] of Object.entries(expected)) {
    if (actual[key] !== expectedValue) {
      throw new Error(
        `Published-style matrix ${label} expected ${key}=${JSON.stringify(expectedValue)}, got ${JSON.stringify(actual[key])}.`,
      );
    }
  }
}

function assertArrayIncludes(label, actual, expectedValues) {
  if (!Array.isArray(actual)) {
    throw new Error(`Published-style matrix ${label} expected an array.`);
  }
  const missing = expectedValues.filter((value) => !actual.includes(value));
  if (missing.length > 0) {
    throw new Error(`Published-style matrix ${label} missing values: ${missing.join(", ")}`);
  }
}

function assertSerializedContains(label, actual, expectedText) {
  const serialized = JSON.stringify(actual);
  if (!serialized.includes(expectedText)) {
    throw new Error(`Published-style matrix ${label} did not contain expected text.`);
  }
}

function assertSerializedExcludes(label, actual, forbiddenText) {
  const serialized = JSON.stringify(actual);
  if (serialized.includes(forbiddenText)) {
    throw new Error(`Published-style matrix ${label} contained forbidden text.`);
  }
}

function assertNoRawSecrets(label, actual, rawSecrets) {
  const serialized = JSON.stringify(actual);
  for (const secret of rawSecrets) {
    if (serialized.includes(secret.value)) {
      throw new Error(`Published-style matrix ${label} leaked raw ${secret.label}.`);
    }
  }
}
